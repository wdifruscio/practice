Node InsertNth(Node head, int data, int position) {
    Node newNode = new Node(); 
    newNode.data = data;
    
    if (head==null) {
        return newNode;        
    } else if (position == 0) {
        newNode.next = head; 
        return newNode;  
    } else {
        int currPosition = 0;
        Node current = head;
        while (currPosition < position -1 && current.next != null) {
            current = current.next;        
            currPosition++;       
        }

        Node nextNode = current.next;
        current.next = newNode;
        current = current.next;
        current.next = nextNode;
    }   
    return head;
}
